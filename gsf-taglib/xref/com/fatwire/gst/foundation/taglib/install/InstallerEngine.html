93      public int getInstallStatus() {
94          savePubid();
95          int status = STATUS_OK;
96          status += _checkPublication();
97          status += _checkTagRegistry();
98          status += _checkUrlRegistry();
99          status += _checkFlexFamily();
100         status += _checkFlexAttributes();
101         status += _checkAttributesInFamily();
102         status += _checkDefinitions();
103         status += _checkAssembler();
104         status += _checkPageRef();
105         status += _checkUserInSite();
106         status += _checkTreetabsInSite();
107         status += _checkCSElementSiteEntryInSite();
108         status += _checkGSTDispatcher();
109         restorePubid();
110         return status;
111     }

113     private int _checkPublication() {
114         //        <PUBLICATION.LOAD NAME="pubtest" FIELD="name" VALUE="GST"/>
115         PublicationLoad load = new PublicationLoad();
116         load.setName("pubtest");
117         load.setField("name");
118         load.setValue("GST");
119         load.execute(ics);
120         return ics.GetErrno() == -101 ? STATUS_NO_PUBLICATION : STATUS_OK;
121     }

123     private int _checkTagRegistry() {
124         if (!new TableTaggingServiceImpl(ics).isInstalled()) return STATUS_NO_TAG_REGISTRY;
125         if (!new TaggedAssetEventListener().isInstalled(ics)) return STATUS_NO_TAG_REGISTRY;
126         if (!new CacheMgrTaggedAssetEventListener().isInstalled(ics)) return STATUS_NO_TAG_REGISTRY;
127         return STATUS_OK;
128     }

132         if (!new WraAssetEventListener().isInstalled(ics)) return STATUS_NO_URL_REGISTRY;

136     private int _checkFlexFamily() {
137         for (String tbl : Arrays.asList("GSTFilter", "GSTAttribute", "GSTPDefinition", "GSTDefinition", "GSTParent", "GSTVirtualWebroot", "GSTAlias", "GSTProperty")) {

174     private int _checkAssembler() {
175         int result = STATUS_NO_ASSEMBLER;
176         try {
177             String srProps = ics.getIServlet().getServlet().getServletContext().getRealPath("/WEB-INF/classes/ServletRequest.properties");
178             InputStream in = new FileInputStream(srProps);
179             Properties prop = new Properties();
180             prop.load(in);
181             in.close();
182             int i = 1;
183             for (; ; ) {
184 
185                 String classname = prop.getProperty("uri.assembler." + i + ".classname");
186                 //out.write(classname +"<br/>");
187                 if (classname == null || classname.trim().length() == 0) {
188                     i--;
189                     break;
190                 }
191 
192                 if ("com.fatwire.gst.foundation.url.WraPathAssembler".equals(classname)) {
193                     result = STATUS_OK;
194                 }
195 
196                 i++;
197             }
198         } catch (IOException e) {
199             LOG.error("Failure checking installer status: " + e, e);
200             result = STATUS_NO_ASSEMBLER;
201 
202         }
203         return result;
204     }

219             }
220         } catch (IOException e) {
221             LOG.error("Failure installing PageRef: " + e, e);
222         }
223         return status;

226     private int _checkUserInSite() {
227         UserManager um = (UserManager) SessionFactory.getSession(ics).getManager(UserManager.class.getName());
228         try {
229             for (User u : um.read(Arrays.asList(ics.GetSSVar("username")))) {
230                 List<String> roles = u.getRoles("GST");
231                 if (roles == null) return STATUS_NO_USER_IN_SITE;
232                 if (roles.containsAll(Arrays.asList("GeneralAdmin", "AdvancedUser"))) {
233                     return STATUS_OK;
234                 } else {
235                     return STATUS_NO_USER_IN_SITE;
236                 }
237             }
238             return STATUS_NO_USER_IN_SITE;
239         } catch (SiteAccessException e) {
240             LOG.error("Exception accessing user data for site: " + e, e);
241             return STATUS_NO_USER_IN_SITE;
242         }

245     /**
246      * Check if TreeTabs have GST site enabled or not. If tab is missing, that's ok (i.e. for Bookmarks or Active List tabs).
247      * If all TreeTabs have the GST site enabled, then return STATUS_OK. Otherwise return STATUS_NO_TREETABS_IN_SITE.
248      * @return
249      */
250     private int _checkTreetabsInSite() {
251         LOG.debug("Checking for GST Site TreeTabs");
252         FTValList list;
253         for(String t : GST_SITE_TREETABS) {
254             LOG.debug("Checking if tab '" + t + "' is enabled for GST site.");
255 //            <TTM.LOAD OBJVARNAME="tab" NAME="GSTSiteTabs.ITEM" />
256 //            <TREETAB.GETID NAME="tab" VARNAME="tabid"/>
257 //            <TREETAB.GETSITES NAME="tab" OBJVARNAME="tsites"/>
258 //            <SITELIST.HASSITE NAME="tsites" PUBID="1328078963506" VARNAME="hasSite"/>
259             list = new FTValList();
260             list.setValString("OBJVARNAME", "tab");
261             list.setValString("NAME", t);
262             ics.runTag("TTM.LOAD", list);
263 
264             if("0".equals(ics.GetVar("errno"))) {
265                 // Do we need to check for tab id here (to confirm it exists)?
266                 ics.SetVar("tabid", "");
267                 list = new FTValList();
268                 list.setValString("NAME", "tab");
269                 list.setValString("VARNAME", "tabid");
270                 ics.runTag("TREETAB.GETID", list);
271                 if(Utilities.goodString(ics.GetVar("tabid"))) {
272                     LOG.debug("Tab '" + t + "' exists. Checking for GST site.");
273                     list = new FTValList();
274                     list.setValString("NAME", "tab");
275                     list.setValString("OBJVARNAME", "tsites");
276                     ics.runTag("TREETAB.GETSITES", list);
277                     list = new FTValList();
278                     list.setValString("NAME", "tsites");
279                     list.setValString("PUBID", ics.GetSSVar("pubid"));
280                     list.setValString("VARNAME", "hasSite");
281                     ics.runTag("SITELIST.HASSITE", list);
282                     if("true".equals(ics.GetVar("hasSite"))) {
283                         LOG.debug("TreeTab '" + t + "' has GST site enabled.");
284                     }
285                     else {
286                         LOG.debug("TreeTab '" + t + "' exists but does NOT have GST site enabled.");
287                         return STATUS_NO_TREETABS_IN_SITE;
288                     }
289                 }
290             }
291             else {
292                 LOG.debug("Unable to load TreeTab '" + t + "', skipping tab (this is OK). Errno=" + ics.GetVar("errno"));
293             }
294         }
295         LOG.debug("All TreeTabs checked have GST site enabled. Returning STATUS_OK.");
296         return STATUS_OK;
297     }

299     /**
300      * If CSElement and SiteEntry asset types are disabled in GST site, or if the GST/Dispatcher CSElement or SiteEntry
301      * is not shared with GST Site, or if their Start Menus don't exist for the GST site,
302      * then return status STATUS_CSE_SE_DISABLED_IN_SITE.
303      * @return
304      */
305     private int _checkCSElementSiteEntryInSite() {
306         LOG.debug("Starting checks for CSElement/SiteEntry.");
307         String[] assetTypes = {"CSElement", "SiteEntry"};
308         String[][] startMenuTypes = {{"ContentForm","New "}, {"Search","Find "}};
309         FTValList list;
310 
311         // check Start Menus
312         for(String at : assetTypes) {
313             for(String[] sm : startMenuTypes) {
314                 LOG.debug("Doing Start Menu check for AssetType=" + at + ", StartMenu type=" + sm[0]);
315 //            <STARTMENU.LOAD NAME="Find AssetTypes.ITEM, GST" ITEMTYPE="Search" OBJVARNAME="miTest"/>
316 //            <STARTMENUITEM.GETID NAME="miTest" VARNAME="miExists"/>
317                 list = new FTValList();
318                 list.setValString("NAME", (sm[1] + at + ", GST"));
319                 list.setValString("ITEMTYPE", sm[0]);
320                 list.setValString("OBJVARNAME", "smTest");
321                 ics.runTag("STARTMENU.LOAD", list);
322                 LOG.debug("Start Menu loaded. Errno=" + ics.GetVar("errno"));
323                 list = new FTValList();
324                 list.setValString("NAME", "smTest");
325                 list.setValString("VARNAME", "smId");
326                 ics.runTag("STARTMENUITEM.GETID", list);
327                 if(Utilities.goodString(ics.GetVar("smId"))) {
328                     LOG.debug("StartMenu found for AssetType=" + at + ", StartMenu type=" + sm[0] + ", id=" + ics.GetVar("smId"));
329 //                    <STARTMENUITEM.GETSITES NAME="smTest" OBJVARNAME="smSiteList" />
330 //                    <SITELIST.HASSITE NAME="smSiteList" PUBID="1328078963506" VARNAME="hasSite"/>
331                     list = new FTValList();
332                     list.setValString("NAME", "smTest");
333                     list.setValString("OBJVARNAME", "smSiteList");
334                     ics.runTag("STARTMENUITEM.GETLEGALSITES", list);
335                     boolean hasSite = siteListHasSite("smSiteList", ics.GetSSVar("pubid"));
336                     if(hasSite) {
337                         LOG.debug("StartMenu '" + (sm[1] + at + ", GST") + "' has GST site enabled.");
338                     }
339                     else {
340                         LOG.debug("StartMenu '" + (sm[1] + at + ", GST") + "' exists but does not have GST site enabled.");
341                         return STATUS_CSE_SE_DISABLED_IN_SITE;
342                     }
343                 }
344                 else {
345                     LOG.debug("StartMenu not found for AssetType=" + at + ", StartMenu type=" + sm[0]);
346                     return STATUS_CSE_SE_DISABLED_IN_SITE;
347                 }
348             }
349         }
350 
351         // check if CSElement, SiteEntry Asset Types are enabled
352         LOG.debug("Done checking StartMenus. Checking if AssetTypes are enabled in GST site.");
353         for(String at : assetTypes) {
354             FTValList args = new FTValList();
355             args.setValString("upcommand", "IsAssetTypeEnabled");
356             args.setValString("assettype", at);
357             args.setValString("pubid", ics.GetSSVar("pubid"));
358             ics.CallElement("OpenMarket/Xcelerate/Actions/AssetMgt/EnableAssetTypePub", args);
359             if("true".equals(ics.GetVar("IsAuthorized"))) {
360                 LOG.debug("GST Site is enabled for AssetType=" + at);
361             }
362             else {
363                 LOG.debug("GST Site is not enabled for AssetType=" + at + ", IsAuthorized=" + ics.GetVar("IsAuthorized"));
364                 return STATUS_CSE_SE_DISABLED_IN_SITE;
365             }
366         }
367 
368         LOG.debug("Finished checks for CSElement/SiteEntry. Returning STATUS_OK.");
369         return STATUS_OK;
370     }

372     /**
373      * Check if a CSElement and SiteEntry named "GST/Dispatcher" exist and are shared with the GST site.
374      * @return
375      */
376     private int _checkGSTDispatcher() {
377         String[] assetTypes = {"CSElement", "SiteEntry"};
378         for(String type : assetTypes) {
379             LOG.debug("Checking if " + type + " 'GST/Dispatcher' exists.");
380 
381             AssetId aid = AssetList.lookupAssetId(ics, type, "GST/Dispatcher");
382             if (aid == null) {
383                 LOG.debug("Could not find " + type + " 'GST/Dispatcher'.");
384                 return STATUS_GST_DISPATCHER_MISSING;
385             }
386 
387             LOG.debug("Found " + type + " 'GST/Dispatcher' with id=" + aid.getId() + ". Checking if it's shared with GST site (pubid " + ics.GetSSVar("pubid") + ")...");
388             FTValList args = new FTValList();
389             args.put("TYPE", type);
390             args.put("OBJECTID", Long.toString(aid.getId()));
391             args.put("LIST", "cseSiteList");
392             args.put("PUBID", ics.GetSSVar("pubid"));
393             ics.runTag("ASSET.SITES", args);
394             if(Utilities.goodString(ics.GetVar("errno"))) {
395                 try {
396                     int errno = Integer.parseInt(ics.GetVar("errno"));
397                     if(errno >= 0) {
398                         LOG.debug(type + " GST/Dispatcher is enabled in the GST site.");
399                     }
400                     else {
401                         LOG.debug(type + " GST/Dispatcher is NOT enabled in the GST site. Errno=" + ics.GetVar("errno"));
402                         return STATUS_GST_DISPATCHER_MISSING;
403                     }
404                 } catch (NumberFormatException e) {
405                     LOG.debug("Error parsing errno in GST/Dispatcher ASSET.SITES check.");
406                     return STATUS_GST_DISPATCHER_MISSING;
407                 }
408             }
409             else {
410                 LOG.debug(type + " GST/Dispatcher is NOT enabled in the GST site.");
411                 return STATUS_GST_DISPATCHER_MISSING;
412             }
413         }
414 
415         return STATUS_OK;
416     }

434     /**
435      * Perform the install.  Note this will only install components
436      * that are not already properly installed, and this determination is made
437      * based on the install status.
438      *
439      * @param status current install status code
440      */
441     protected void doInstall(int status) {
442         savePubid();
443         LOG.info("Installing missing GSF components");
444         if (isInstallNeeded(STATUS_NO_PUBLICATION, status)) installPublication();
445         if (isInstallNeeded(STATUS_NO_TAG_REGISTRY, status)) installTagRegistry();
446         if (isInstallNeeded(STATUS_NO_URL_REGISTRY, status)) installUrlRegistry();
447         if (isInstallNeeded(STATUS_NO_FLEX_FAMILY, status)) installFlexFamily();
448         if (isInstallNeeded(STATUS_NO_FLEX_ATTRIBUTES, status)) installFlexAttributes();
449         if (isInstallNeeded(STATUS_NO_ATTRIBUTES_IN_FAMILY, status)) installAttributesIntoFamily();
450         if (isInstallNeeded(STATUS_NO_DEFINITIONS, status)) installDefinitions();
451         if (isInstallNeeded(STATUS_NO_ASSEMBLER, status)) installAssembler();
452         if (isInstallNeeded(STATUS_NO_PAGEREF, status)) installPageRef();
453         if (isInstallNeeded(STATUS_NO_USER_IN_SITE, status)) installUserIntoSite();
454         if (isInstallNeeded(STATUS_NO_TREETABS_IN_SITE, status)) installTreetabsIntoSite();
455         if (isInstallNeeded(STATUS_CSE_SE_DISABLED_IN_SITE, status)) installCSElementSiteEntryInSite();
456         if (isInstallNeeded(STATUS_GST_DISPATCHER_MISSING, status)) installGSTDispatcher();
457         LOG.info("GSF missing component install complete.");
458         restorePubid();
459     }
460 
461     final boolean isInstallNeeded(int componentStatus, int systemStatus) {
462         return (systemStatus & componentStatus) == componentStatus;
463     }

465     private void savePubid() {
466         String s = ics.GetSSVar("pubid");
467         if (StringUtils.isNotBlank(s)) {
468             ics.SetVar("oldpubid", s);
469             ics.RemoveSSVar("pubid");
470         }
471 
472         PublicationLoad load = new PublicationLoad();
473         load.setName("pubload");
474         load.setField("name");
475         load.setValue("GST");
476         load.execute(ics);
477         PublicationGet pGet = new PublicationGet();
478         pGet.setName("pubload");
479         pGet.setField("id");
480         pGet.setOutput("pubid");
481         pGet.execute(ics);
482         ics.SetSSVar("pubid", ics.GetVar("pubid"));
483     }
484 
485     private void restorePubid() {
486         String s = ics.GetVar("oldpubid");
487         if (StringUtils.isNotBlank(s)) {
488             ics.RemoveVar("oldpubid");
489             ics.SetSSVar("pubid", s);
490         }
491     }

493     void installPublication() {
494         LOG.info("Installing Publication...");
495         //     <SETVAR NAME="Publication:name" VALUE="GST"/>
496         //     <SETVAR NAME="Publication:description" VALUE="GST site"/>
497         //     <SETVAR NAME="Publication:cs_preview" VALUE="Variables.empty"/>
498         ics.SetVar("Publication:name", "GST");
499         ics.SetVar("Publicatin:description", "GST Site");
500         ics.SetVar("Publication:cs_preview", "");
501         //     <SETVAR NAME="errno" VALUE="0"/>
502         ics.ClearErrno();
503         //     <PUBLICATION.CREATE NAME="publication"/>
504         PublicationCreate pc = new PublicationCreate();
505         pc.setName("publication");
506         pc.execute(ics);
507         //     <PUBLICATION.GATHER NAME="publication" PREFIX="Publication"/>
508         PublicationGather pg = new PublicationGather();
509         pg.setName("publication");
510         pg.setPrefix("Publication");
511         pg.execute(ics);
512         //     <PUBLICATION.SAVE NAME="publication"/>
513         PublicationSave ps = new PublicationSave();
514         ps.setName("publication");
515         ps.execute(ics);
516         //     <PUBLICATION.GET NAME="publication" FIELD="id" OUTPUT="pubid"/>
517         PublicationGet pGet = new PublicationGet();
518         pGet.setName("publication");
519         pGet.setField("id");
520         pGet.setOutput("pubid");
521         //     <h2>Created GST site with id: <CSVAR NAME="Variables.pubid"/></h2>
522         LOG.info("...publication install complete.  Pubid: " + ics.GetVar("pubid"));
523     }

696     /**
