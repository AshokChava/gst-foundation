53  /**
54   * Factory implementation that works with a method naming convention to create
55   * objects. Objects are created in a delegated factory method. The delegated
56   * method is found by looking for a method that is prefixed by 'create' and then
57   * the the simple name of the class (classname without package prefix).
58   * <p>
59   * For instance to create a object of class 'com.bar.Foo' if will look for a
60   * method <tt>public Foo createFoo(ICS ics);</tt>. The method has to be public
61   * and has to accept one argument of type ICS.
62   * </p>
63   * 
64   * 
65   * @author Dolf.Dijkstra
66   * @since Apr 20, 2011
67   * 
68   */
69  public class IcsBackedObjectFactoryTemplate implements Factory {
70      protected static final Log LOG = LogUtil.getLog(IcsBackedObjectFactoryTemplate.class);
71      private final ICS ics;
72      private final Map<String, Object> objectCache = new HashMap<String, Object>();
73  
74      /**
75       * Constructor.
76       * 
77       * @param ics the Content Server context
78       */
79      public IcsBackedObjectFactoryTemplate(final ICS ics) {
80          super();
81          this.ics = ics;
82      }
83  
84      /*
85       * (non-Javadoc)
86       * 
87       * @see
88       * com.fatwire.gst.foundation.controller.action.AnnotationInjector.Factory
89       * #getObject(java.lang.String, java.lang.Class)
90       */
91      public final <T> T getObject(final String name, final Class<T> fieldType) {
92  
93          return locate(fieldType, ics);
94      }
95  
96      /**
97       * Should the created object be cached on the ICS scope.
98       * 
99       * @param c
100      * @return true is object should be cached locally
101      */
102     public boolean shouldCache(final Class<?> c) {
103         // don't cache the model as this is bound to the jsp page context and
104         // not
105         // to ICS. It would leak variables into other elements if we allowed it
106         // to cache.
107         // TODO:medium, figure out if this should be done more elegantly. It
108         // seems that scoping logic is
109         // brought into the factory, that might be a bad thing.
110         if (Model.class.isAssignableFrom(c)) {
111             return false;
112         }
113 
114         return true;
115     }
116 
117     @SuppressWarnings("unchecked")
118     protected <T> T locate(final Class<T> c, final ICS ics) {
119         if (ICS.class.isAssignableFrom(c)) {
120             return (T) ics;
121         }
122         if (c.isArray()) {
123             throw new IllegalArgumentException("Arrays are not supported");
124         }
125         final String name = c.getSimpleName();
126 
127         if (StringUtils.isBlank(name)) {
128             return null;
129         }
130         Object o = objectCache.get(name);
131         if (o == null) {
132 
133             try {
134                 // TODO: medium: check for other method signatures
135                 Method m;
136                 m = getClass().getMethod("create" + name, ICS.class);
137                 if (m != null) {
138                     o = m.invoke(this, ics);
139                 }
140             } catch (final NoSuchMethodException e) {
141                 try {
142                     LOG.debug("Could not create  a " + c.getName() + " via a Template method, trying via constructor.");
143                     final Constructor<T> constr = c.getConstructor(ICS.class);
144                     o = constr.newInstance(ics);
145                 } catch (final RuntimeException e1) {
146                     throw e1;
147                 } catch (final Exception e1) {
148                     throw new RuntimeException(e1);
149                 }
150             } catch (final RuntimeException e) {
151                 throw e;
152             } catch (final Exception e) {
153                 throw new RuntimeException(e);
154             }
155             if (shouldCache(c)) {
156                 objectCache.put(c.getName(), o);
157             }
158         }
159         return (T) o;
160     }
